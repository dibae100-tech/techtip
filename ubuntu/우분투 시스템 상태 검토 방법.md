### üì¶ 10. system_check_report.sh
    #!/bin/bash
    
    # Ï†ÄÏû• Í≤ΩÎ°ú ÏÑ§Ï†ï
    REPORT_DIR="$HOME/system_check"
    mkdir -p "$REPORT_DIR"  # Ìè¥Îçî ÏóÜÏúºÎ©¥ ÏÉùÏÑ±
    REPORT_FILE="$REPORT_DIR/system_report_$(date +%Y%m%d_%H%M%S).txt"
    
    # Î≥¥Í≥†ÏÑú ÏûëÏÑ± ÏãúÏûë
    echo "========================================" | tee "$REPORT_FILE"
    echo "üß† SYSTEM HEALTH REPORT - $(date)" | tee -a "$REPORT_FILE"
    echo "========================================" | tee -a "$REPORT_FILE"
    
    echo -e "\nüîß Failed System Services:" | tee -a "$REPORT_FILE"
    systemctl --failed | tee -a "$REPORT_FILE"
    
    echo -e "\nüîç Recent Critical & Error Logs (journalctl -p 3 -xb):" | tee -a "$REPORT_FILE"
    journalctl -p 3 -xb -n 50 | tee -a "$REPORT_FILE"
    
    echo -e "\n‚ö†Ô∏è Kernel Error/Warning Messages (dmesg):" | tee -a "$REPORT_FILE"
    dmesg --level=err,warn | tail -n 50 | tee -a "$REPORT_FILE"
    
    echo -e "\nüíæ Disk Usage (df -h):" | tee -a "$REPORT_FILE"
    df -h | tee -a "$REPORT_FILE"
    
    echo -e "\nüßÆ Memory Usage (free -h):" | tee -a "$REPORT_FILE"
    free -h | tee -a "$REPORT_FILE"
    
    echo -e "\nüì¶ Upgradable Packages:" | tee -a "$REPORT_FILE"
    apt list --upgradable 2>/dev/null | tee -a "$REPORT_FILE"
    
    echo -e "\nüìÅ /var/log Directory Usage:" | tee -a "$REPORT_FILE"
    sudo du -sh /var/log/* | sort -h | tee -a "$REPORT_FILE"
    
    echo -e "\nüîå Open Ports & Listening Services (ss):" | tee -a "$REPORT_FILE"
    ss -tulpn | tee -a "$REPORT_FILE"
    
    echo -e "\nüì° Network Interfaces (ip a):" | tee -a "$REPORT_FILE"
    ip a | tee -a "$REPORT_FILE"
    
    echo -e "\nüßæ Fstab Entries (if any mount errors):" | tee -a "$REPORT_FILE"
    cat /etc/fstab | grep -v '^#' | grep -v '^$' | tee -a "$REPORT_FILE"
    
    echo -e "\n‚úÖ System Summary Finished. Report saved to $REPORT_FILE"
